#include<stdio.h>
void main()
{
    int a,b,c;
    int t;
    scanf("%d",&t);
        while(t--)
        {   
            scanf("%d %d %d",&a,&b,&c);
            if (a==b&&b!=c)
            {
                printf("%d\n",c);
            }
            else if (a==c&&b!=c)
            {
                printf("%d\n",b);
            }
            else
            {
                printf("%d\n",a);
            }
            t>0; 
        }
      }
